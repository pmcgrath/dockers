See http://docs.docker.io/en/latest/installation/ubuntulinux/
For an existing ubuntu 12.04
	# Upgrade to a 3.8 kernel which is a requirement
	sudo apt-get update
	sudo apt-get -y install linux-image-generic-lts-raring linux-headers-generic-lts-raring
	sudo reboot

	# Install docker
	sudo apt-key adv --keyserver keyserver.ubuntu.com --recv-keys 36A1D7869245C8950F966E92D8576A8BA88D21E9
	sudo sh -c "echo deb http://get.docker.io/ubuntu docker main > /etc/apt/sources.list.d/docker.list"
	sudo apt-get update
	sudo apt-get -y install lxc-docker

	# Verify docker version
	sudo docker version
